```shell

# --env=pro指定环境，可选，如果不写默认是--env=test
```

## 用例数据格式说明
```ts
type casesItem = {
  /** 用例标题 */
  title: string,
  /** 是否在生产环境运行 */
  runPro?: "true" | "false",
  /** 路径参数 activity/changeActivity/{id} 传入{id: 1}输出activity/changeActivity/{1}*/
  pathParams?: Dictionary,
  /** 接口参数 */
  data?: Dictionary,
  /** 期望的结果 */
  expected: {
      header: Dictionary
      body?: Dictionary
  }
},

type DataCases = {
  /** 用例分类 */
  feature: string,
  /** 用例函数名称 */
  funcNameSpace: string,
  /** 重要等级 */
  funcNameSpace: 'BLOCKER' | 'CRITICAL' | 'NORMAL' | 'MINOR' | 'TRIVIAL',
  /** 用例数据相对路径 */
  casePath: string,
  /** 接口路径 */
  path: string
  /** 详细用例数据 */
  casesList: casesItem[]
}

```
